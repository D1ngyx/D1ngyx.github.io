class Dsu {
private:
    vector<int> root_of_;
public:
    Dsu(int size) {
        root_of_.resize(size);
        std::iota(root_of_.begin(), root_of_.end(), 0);
    }
    int find(int x) {
        if (root_of_[x] != x) {
            root_of_[x] = find(root_of_[x]);
        }
        return root_of_[x];
    }
    void unite(int x, int y) {
        root_of_[find(x)] = find(y);
    }
    bool same(int x, int y) {
        return find(x) == find(y);
    }
};
